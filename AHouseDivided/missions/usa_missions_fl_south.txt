	}

	mission_fl_south_stabilize_south_florida = {
		icon = mission_alliances
		position = 3
		required_missions = { mission_fl_south_dominate_south_florida }
		trigger = {
			area_fl_gold_coast = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
			area_fl_southwest = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
			area_fl_sarasota = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
			area_fl_treasure_coast = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
			area_fl_heartland = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
 		}
		effect = {
			add_dip_power = 100
			add_adm_power = 100
			add_prestige_or_monarch_power = { amount = 10 }
		}
		provinces_to_highlight = {
			OR = {
				area = area_fl_gold_coast
				area = area_fl_southwest
				area = area_fl_sarasota
				area = area_fl_treasure_coast
				area = area_fl_heartland
			}
			OR = {
				NOT { country_or_non_sovereign_subject_holds = ROOT }
				devastation = 1
				unrest = 10
				local_autonomy = 50
			}
		}
	}

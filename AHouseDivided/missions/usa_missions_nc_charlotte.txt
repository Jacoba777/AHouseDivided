	generic = no
	ai = yes

	potential = { tag = CLT }

	mission_nc_charlotte_consolidate_charlotte = {
		icon = mission_clt
		position = 1
		required_missions = {  }
		trigger = {
            NOT = { num_of_subjects = 1 }
			area_nc_charlotte = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_nc_north_charlotte = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_nc_east_charlotte = { type = all  country_or_non_sovereign_subject_holds = ROOT }
 		}
		effect = {
			area_nc_mount_airy = { add_permanent_claim = ROOT }
			area_nc_winstone = { add_permanent_claim = ROOT }
			area_nc_greensboro = { add_permanent_claim = ROOT }
			area_nc_pinehurst = { add_permanent_claim = ROOT }
			area_nc_fayetteville = { add_permanent_claim = ROOT }
		}
		provinces_to_highlight = {
			OR = {

				area = area_nc_charlotte
				area = area_nc_north_charlotte
				area = area_nc_east_charlotte
			}
			NOT = { owned_by = ROOT }
		}
	}

	mission_nc_charlotte_invade_greensboro = {
		icon = mission_gnr
		position = 2
		required_missions = { mission_nc_charlotte_consolidate_charlotte }
		trigger = {

		effect = {

		position = 3

		trigger = {
			area_nc_winstone = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT

			}

				type = all

					NOT = { local_autonomy = 50 }

			area_nc_west_research_triangle = { add_permanent_claim = ROOT }
			area_nc_raleigh = { add_permanent_claim = ROOT }
		}
		provinces_to_highlight = {
			OR = {
				area = area_nc_winstone
				area = area_nc_greensboro
			}
			OR = {
				NOT { country_or_non_sovereign_subject_holds = ROOT }
				devastation = 1
				unrest = 10
				local_autonomy = 50
			}
		}
	}

			}
			NOT = { owned_by = ROOT }
		}
	}

	mission_nc_charlotte_bring_order_to_raleigh = {
		icon = mission_alliances
		position = 5
		required_missions = { mission_nc_charlotte_capture_the_old_capital }
		trigger = {
			area_nc_west_research_triangle = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
			area_nc_raleigh = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
 		}
		effect = {

		}
		provinces_to_highlight = {

				area = area_nc_west_research_triangle
				area = area_nc_raleigh

				NOT { country_or_non_sovereign_subject_holds = ROOT }
				devastation = 1
				unrest = 10
				local_autonomy = 50
			}
		}
	}

	mission_nc_charlotte_build_a_navy = {
		icon = mission_establish_high_seas_navy
		position = 6
		required_missions = {  }
		trigger = {
            navy_size_percentage = 1
            has_flagship = yes
            OR = {
                num_of_galley = 20
                num_of_heavy_ship = 10
            }
 		}
		effect = {
            if = {
                limit = { NOT = { navy_tradition = 40 } }
                create_admiral = { tradition = 40 }
            }
            else_if = {
                limit = { NOT = { navy_tradition = 60 } }
                create_admiral = { tradition = 60 }
            }
            else_if = {
                limit = { NOT = { navy_tradition = 80 } }
                create_admiral = { tradition = 80 }
            }
            else = {
                create_admiral = { tradition = 100 }
            }
			add_country_modifier = {
				name = country_modifier_naval_enthusiasm
				duration = 9125  # 25 years
			}
		}

	}
}

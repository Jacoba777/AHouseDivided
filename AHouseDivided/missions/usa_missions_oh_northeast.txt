	mission_oh_northeast_denounce_toledo = {
		icon = mission_luther_theses
		position = 2
		required_missions = {  }
		trigger = {
			OR = {
				is_rival = TLD
				AND = {
					exists = TLD
					military_strength = {
						who = TLD
						value = 2.0
					}
				}
				power_projection = 80
				tag = TLD
			}
			OR = {
				TLD = {
					has_opinion_modifier = {
						who = ROOT
						modifier = scornfully_insulted
					}
				}
				TLD = {
					has_opinion_modifier = {
						who = ROOT
						modifier = insulted
					}
				}
				NOT = { exists = TLD }
				tag = TLD
			}
 		}
		effect = {
			area_oh_sandusky = { add_permanent_claim = ROOT }
			area_oh_toledo = { add_permanent_claim = ROOT }
		}

	}

	mission_oh_northeast_dominate_dayton = {
		icon = mission_day
		position = 7
		required_missions = { mission_oh_northeast_integrate_the_northwest }
		trigger = {

 		}

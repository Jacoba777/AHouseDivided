		position = 3
		required_missions = { mission_va_chesapeake_peninsular_warfare }
		trigger = {
            navy_size_percentage = 1
            has_flagship = yes
            OR = {
                num_of_galley = 20
                num_of_heavy_ship = 10
            }
 		}
		effect = {
            if = {
                limit = { NOT = { navy_tradition = 40 } }
                create_admiral = { tradition = 40 }
            }
            else_if = {
                limit = { NOT = { navy_tradition = 60 } }
                create_admiral = { tradition = 60 }
            }
            else_if = {
                limit = { NOT = { navy_tradition = 80 } }
                create_admiral = { tradition = 80 }
            }
            else = {
                create_admiral = { tradition = 100 }
            }
			add_country_modifier = {
				name = country_modifier_naval_enthusiasm
				duration = 9125  # 25 years
			}
		}

	}

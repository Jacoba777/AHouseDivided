	mission_sc_lowcountry_going_coastal = {
		icon = mission_unite_home_region
		position = 2
		required_missions = { mission_sc_lowcountry_fielding_an_army }
		trigger = {
			area_sc_charleston = {
				type = all  
				OR = {

 		}
		effect = {
			add_country_modifier = {

			}
		}

			OR = {
				area = area_sc_charleston
				area = area_sc_lowcountry
			}
			NOT = { owned_by = ROOT }
			is_city = yes
		}
	}

				}
			}
 		}
		effect = {
			area_sc_columbia = { add_permanent_claim = ROOT }

	mission_il_west_the_elephant_in_the_room = {
		icon = mission_luther_theses
		position = 5
		required_missions = { mission_il_west_develop_springfield }
		trigger = {
			OR = {
				is_rival = CCG
				AND = {
					exists = CCG
					military_strength = {
						who = CCG
						value = 2.0
					}
				}
				power_projection = 80
				tag = CCG
			}
			OR = {
				CCG = {
					has_opinion_modifier = {
						who = ROOT
						modifier = scornfully_insulted
					}
				}
				CCG = {
					has_opinion_modifier = {
						who = ROOT
						modifier = insulted
					}
				}
				NOT = { exists = CCG }
				tag = CCG
			}
 		}
		effect = {
			area_il_ottawa = { add_permanent_claim = ROOT }
			area_il_chicago = { add_permanent_claim = ROOT }
			area_il_south_chicago = { add_permanent_claim = ROOT }
		}

	}

	mission_il_west_the_fate_of_windy_city = {
		icon = mission_ccg
		position = 6
		required_missions = { mission_il_west_the_elephant_in_the_room }
		trigger = {
			area_il_ottawa = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_il_chicago = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_il_south_chicago = { type = all  country_or_non_sovereign_subject_holds = ROOT }
 		}

			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
	}

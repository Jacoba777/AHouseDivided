missions_sc_midlands_slot1 = {
	slot = 1
	generic = no
	ai = yes
	has_country_shield = yes
	potential = {
		OR = {
			tag = CUM
			tag = ONG
			tag = NBR
		}
	}

	mission_sc_midlands_challenging_charleston = {
		icon = mission_cls
		position = 3
		required_missions = { mission_sc_midlands_unite_the_midlands }
		trigger = {
			OR = {
				is_rival = CLS
				AND = {
					exists = CLS
					military_strength = {
						who = CLS
						value = 2.0
					}
				}
				power_projection = 80
				tag = CLS
			}
			OR = {
				CLS = {
					has_opinion_modifier = {
						who = ROOT
						modifier = scornfully_insulted
					}
				}
				CLS = {
					has_opinion_modifier = {
						who = ROOT
						modifier = insulted
					}
				}
				NOT = { exists = CLS }
				tag = CLS
			}
 		}
		effect = {
			CLS = { every_owned_province = { add_permanent_claim = ROOT }  }
			area_sc_charleston = { add_permanent_claim = ROOT }
		}

	}

	mission_sc_midlands_the_siege_of_charleston = {
		icon = mission_unite_home_region
		position = 4
		required_missions = { mission_sc_midlands_challenging_charleston }
		trigger = {
			area_sc_charleston = { type = all  country_or_non_sovereign_subject_holds = ROOT }
 		}
		effect = {
			add_country_modifier = {
				name = country_modifier_a_successful_conquest
				duration = 730  # 2 years
			}
		}
		provinces_to_highlight = {
			OR = {
				area = area_sc_charleston
			}
			NOT = { owned_by = ROOT }
		}
	}

	mission_sc_midlands_stabilize_the_lowcountry = {
		icon = mission_alliances
		position = 5
		required_missions = { mission_sc_midlands_the_siege_of_charleston }
		trigger = {
			area_sc_charleston = {
				type = all  
				AND {
					country_or_non_sovereign_subject_holds = ROOT
					NOT = { devastation = 1 }
					NOT = { unrest = 10 }
					NOT = { local_autonomy = 50 }
				}
			}
 		}
		effect = {
			add_dip_power = 75
			add_prestige_or_monarch_power = { amount = 10 }
		}
		provinces_to_highlight = {
			OR = {
				area = area_sc_charleston
			}
			OR = {
				NOT { country_or_non_sovereign_subject_holds = ROOT }
				devastation = 1
				unrest = 10
				local_autonomy = 50
			}
		}
	}
}

            army_size_percentage = 1
 		}
		effect = {
			area_sc_columbia = { add_permanent_claim = ROOT }
			area_sc_thoroughbred = { add_permanent_claim = ROOT }
			area_sc_newberry = { add_permanent_claim = ROOT }
		}

	}

	mission_sc_midlands_unite_the_midlands = {
		icon = mission_alliances
		position = 2
		required_missions = { mission_sc_midlands_fielding_an_army }
		trigger = {
            NOT = { num_of_subjects = 1 }
			area_sc_columbia = {
				type = all  
				OR = {
					AND {
						country_or_non_sovereign_subject_holds = ROOT
						NOT = { devastation = 1 }
						NOT = { unrest = 10 }
						NOT = { local_autonomy = 50 }
					}
					is_city = no
				}
			}
			area_sc_thoroughbred = {
				type = all  
				OR = {
					AND {
						country_or_non_sovereign_subject_holds = ROOT
						NOT = { devastation = 1 }
						NOT = { unrest = 10 }
						NOT = { local_autonomy = 50 }
					}
					is_city = no
				}
			}
			area_sc_newberry = {
				type = all  
				OR = {
					AND {
						country_or_non_sovereign_subject_holds = ROOT
						NOT = { devastation = 1 }
						NOT = { unrest = 10 }
						NOT = { local_autonomy = 50 }
					}
					is_city = no
				}
			}
 		}
		effect = {
			add_legitimacy = 10
			add_meritocracy = 10
			add_republican_tradition = 5
			add_devotion = 10
			add_horde_unity = 10
			add_prestige_or_monarch_power = { amount = 15 }
		}
		provinces_to_highlight = {
			OR = {

				area = area_sc_columbia
				area = area_sc_thoroughbred
				area = area_sc_newberry
			}
			OR = {
				NOT { country_or_non_sovereign_subject_holds = ROOT }
				devastation = 1
				unrest = 10
				local_autonomy = 50
			}
			is_city = yes
		}
	}
}

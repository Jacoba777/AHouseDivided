	mission_nc_piedmont_enemies_with_raleigh = {
		icon = mission_ral
		position = 3
		required_missions = { mission_nc_piedmont_unite_our_people }
		trigger = {
			OR = {
				is_rival = RAL
				AND = {
					exists = RAL
					military_strength = {
						who = RAL
						value = 2.0
					}
				}
				power_projection = 80
				tag = RAL
			}
			OR = {
				RAL = {
					has_opinion_modifier = {
						who = ROOT
						modifier = scornfully_insulted
					}
				}
				RAL = {
					has_opinion_modifier = {
						who = ROOT
						modifier = insulted
					}
				}
				NOT = { exists = RAL }
				tag = RAL
			}
 		}
		effect = {
			area_nc_north_research_triangle = { add_permanent_claim = ROOT }
			area_nc_west_research_triangle = { add_permanent_claim = ROOT }
			area_nc_raleigh = { add_permanent_claim = ROOT }
		}

	}

	mission_nc_piedmont_capture_the_old_capital = {
		icon = mission_unite_home_region
		position = 4
		required_missions = { mission_nc_piedmont_enemies_with_raleigh }
		trigger = {
			area_nc_north_research_triangle = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_nc_west_research_triangle = { type = all  country_or_non_sovereign_subject_holds = ROOT }
			area_nc_raleigh = { type = all  country_or_non_sovereign_subject_holds = ROOT }
 		}
		effect = {
			add_adm_power = 100
			add_prestige_or_monarch_power = { amount = 15 }
		}
		provinces_to_highlight = {
			OR = {
				area = area_nc_north_research_triangle
				area = area_nc_west_research_triangle
				area = area_nc_raleigh
			}
			NOT = { owned_by = ROOT }
		}
	}

                        NOT = { tag = ROOT }
                    }

 		}

		}
